class GameLogic {
  constructor() {
    this.gameState = {
      map: null,
      playerPosition: null,
      hasKey: false,
      isDetected: false,
      playerMoves: 0,
      maxMoves: 20 // Define maximum moves allowable before player loses
    };
    this.initializeGame();
  }

  initializeGame() {
    const selectedMap = this.getRandomMap();
    this.gameState.map = selectedMap.layout;
    this.positionPlayerAndKey();
  }

  getRandomMap() {
    const randomIndex = Math.floor(Math.random() * maps.length);
    return maps[randomIndex];
  }

  positionPlayerAndKey() {
    for (let i = 0; i < this.gameState.map.length; i++) {
      for (let j = 0; j < this.gameState.map[i].length; j++) {
        if (this.gameState.map[i][j] === 'S') {
          this.gameState.playerPosition = { x: i, y: j };
        }
        if (this.gameState.map[i][j] === 'V') {
          this.gameState.keyPosition = { x: i, y: j };
        }
      }
    }
  }

  movePlayer(direction) {
    if (this.gameState.isDetected || this.gameState.playerMoves >= this.gameState.maxMoves) {
      return this.checkGameOver();
    }

    const { x, y } = this.gameState.playerPosition;
    let newX = x;
    let newY = y;

    switch (direction) {
      case 'up':
        newX -= 1;
        break;
      case 'down':
        newX += 1;
        break;
      case 'left':
        newY -= 1;
        break;
      case 'right':
        newY += 1;
        break;
    }

    if (this.canMoveTo(newX, newY)) {
      this.gameState.playerPosition = { x: newX, y: newY };
      this.gameState.playerMoves += 1;

      const currentTile = this.gameState.map[newX][newY];
      if (currentTile === 'V') {
        this.gameState.hasKey = true;
      } else if (currentTile === 'H') {
        this.detectPlayer();
      } else if (currentTile === 'E' && this.gameState.hasKey) {
        return "Win";
      }
    }
  }

  canMoveTo(x, y) {
    if (x < 0 || y < 0 || x >= this.gameState.map.length || y >= this.gameState.map[0].length) {
      return false;
    }
    return this.gameState.map[x][y] !== 'W';
  }

  detectPlayer() {
    this.gameState.isDetected = true;
  }

  checkGameOver() {
    if (this.gameState.isDetected) {
      return "Lose - Detected!";
    }
    if (this.gameState.playerMoves >= this.gameState.maxMoves) {
      return "Lose - Out of moves!";
    }
  }
}

document.addEventListener('DOMContentLoaded', setupGame);

function setupGame() {
    const gameLogic = new GameLogic();
    const playerElement = document.createElement('div');
    playerElement.id = 'player';
    playerElement.style.position = 'absolute';
    playerElement.style.width = '25px';
    playerElement.style.height = '25px';
    playerElement.style.backgroundColor = 'red';
    document.getElementById('game-canvas').appendChild(playerElement);

    updatePlayerPosition(playerElement, gameLogic);

    window.addEventListener('keydown', (event) => {
        const directionMap = {
            'ArrowUp': 'up',
            'ArrowDown': 'down',
            'ArrowLeft': 'left',
            'ArrowRight': 'right'
        };

        const direction = directionMap[event.key];
        if (direction) {
            gameLogic.movePlayer(direction);
            updatePlayerPosition(playerElement, gameLogic);
        }
    });

    let touchStartX = 0;
    let touchStartY = 0;

    window.addEventListener('touchstart', (event) => {
        touchStartX = event.touches[0].clientX;
        touchStartY = event.touches[0].clientY;
    });

    window.addEventListener('touchend', (event) => {
        const touchEndX = event.changedTouches[0].clientX;
        const touchEndY = event.changedTouches[0].clientY;

        const diffX = touchEndX - touchStartX;
        const diffY = touchEndY - touchStartY;

        if (Math.abs(diffX) > Math.abs(diffY) && Math.abs(diffX) > 30) {
            if (diffX > 0) {
                gameLogic.movePlayer('right');
            } else {
                gameLogic.movePlayer('left');
            }
        } else if (Math.abs(diffY) > 30) {
            if (diffY > 0) {
                gameLogic.movePlayer('down');
            } else {
                gameLogic.movePlayer('up');
            }
        }

        updatePlayerPosition(playerElement, gameLogic);
    });
}

function updatePlayerPosition(playerElement, gameLogic) {
    const tileSize = 30; // Assuming each tile is 30x30 pixels
    const { x, y } = gameLogic.gameState.playerPosition;
    
    playerElement.style.transform = `translate(${y * tileSize}px, ${x * tileSize}px)`;
}

document.addEventListener('DOMContentLoaded', setupGame);

function setupGame() {
    const gameLogic = new GameLogic();
    const playerElement = document.createElement('div');
    playerElement.id = 'player';
    playerElement.style.position = 'absolute';
    playerElement.style.width = '25px';
    playerElement.style.height = '25px';
    playerElement.style.backgroundColor = 'red';
    document.getElementById('game-canvas').appendChild(playerElement);

    updatePlayerPosition(playerElement, gameLogic);

    window.addEventListener('keydown', (event) => {
        const directionMap = {
            'ArrowUp': 'up',
            'ArrowDown': 'down',
            'ArrowLeft': 'left',
            'ArrowRight': 'right'
        };

        const direction = directionMap[event.key];
        if (direction) {
            gameLogic.movePlayer(direction);
            updatePlayerPosition(playerElement, gameLogic);
        }
    });

    let touchStartX = 0;
    let touchStartY = 0;

    window.addEventListener('touchstart', (event) => {
        touchStartX = event.touches[0].clientX;
        touchStartY = event.touches[0].clientY;
    });

    window.addEventListener('touchend', (event) => {
        const touchEndX = event.changedTouches[0].clientX;
        const touchEndY = event.changedTouches[0].clientY;

        const diffX = touchEndX - touchStartX;
        const diffY = touchEndY - touchStartY;

        if (Math.abs(diffX) > Math.abs(diffY) && Math.abs(diffX) > 30) {
            if (diffX > 0) {
                gameLogic.movePlayer('right');
            } else {
                gameLogic.movePlayer('left');
            }
        } else if (Math.abs(diffY) > 30) {
            if (diffY > 0) {
                gameLogic.movePlayer('down');
            } else {
                gameLogic.movePlayer('up');
            }
        }

        updatePlayerPosition(playerElement, gameLogic);
    });
}

function updatePlayerPosition(playerElement, gameLogic) {
    const tileSize = 30; // Assuming each tile is 30x30 pixels
    const { x, y } = gameLogic.gameState.playerPosition;
    
    playerElement.style.transform = `translate(${y * tileSize}px, ${x * tileSize}px)`;
}
